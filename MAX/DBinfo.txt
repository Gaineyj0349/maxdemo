Table creations:
CREATE TABLE _table_conflict(
_id INT AUTO_INCREMENT PRIMARY KEY,
_name VARCHAR(400),
_war_occurance TINYTEXT,
_date_begin DATE,
_date_end DATE,
_associations_list VARCHAR(400),
_summary TEXT,
_who_label TINYTEXT,
_sourced_references VARCHAR(400)
)ENGINE InnoDB;

CREATE TABLE _table_war(
_id INT AUTO_INCREMENT PRIMARY KEY,
_name VARCHAR(400),
_associations_list VARCHAR(400),
_date_begin DATE,
_date_end DATE,
_parties_involved VARCHAR(400),
_sourced_references VARCHAR(400)
)ENGINE InnoDB;

CREATE TABLE _table_associations(
_id INT AUTO_INCREMENT PRIMARY KEY,
_name VARCHAR(400),
_type TINYTEXT,
_associations_id INT
)ENGINE InnoDB;

CREATE TABLE _table_person(
_id INT AUTO_INCREMENT PRIMARY KEY,
_name VARCHAR(400),
_date_birth DATE,
_date_death DATE,
_bio TEXT,
_bio_image VARCHAR(400),
_associations_list VARCHAR(400),
_sourced_references VARCHAR(400)
)ENGINE InnoDB;
